1    package com.example.apnuamdavad;
2    
3    import android.content.Context;
4    import android.content.SharedPreferences;
5    
6    import com.example.apnuamdavad.ApiHelper.JsonField;
7    
8    public class UserSessionManager {
9    
10     //define key
11     private  static  final  String MY_PREF="user_pref";
12     private  static  final  String IS_LOGIN="is_login";
13   
14     Context mContext;
15     SharedPreferences mSharedPreferences;
16   
17     private  final SharedPreferences.Editor mEditor;
18   
19     public UserSessionManager(Context mContext) {
20       this.mContext = mContext;
21       mSharedPreferences=mContext.getSharedPreferences(MY_PREF,Context.MODE_PRIVATE);
22       mEditor=mSharedPreferences.edit();
23     }
24   
25     public void setLoginStatus()
26     {
27       mEditor.putBoolean(IS_LOGIN,true).commit();
28     }
29   
30     public boolean getLoginStatus()
31     {
32       return mSharedPreferences.getBoolean(IS_LOGIN,false);
33   
34     }
35   
36     public  void setUserDetails(String strUserId,String strUserName,String strUserEmail,String strUserPass,String strUserNo)
37     {
38       mEditor.putString(JsonField.KEY_USER_ID,strUserId);
39       mEditor.putString(JsonField.KEY_USER_NAME,strUserName);
40       mEditor.putString(JsonField.KEY_USER_EMAIL,strUserEmail);
41       mEditor.putString(JsonField.KEY_USER_PASSWORD,strUserPass);
42       mEditor.putString(JsonField.KEY_USER_MOBILENO,strUserNo);
43       mEditor.commit();
44   
45     }
46     public void logout()
47     {
48       mEditor.remove(JsonField.KEY_USER_ID);
49       mEditor.remove(JsonField.KEY_USER_NAME);
50       mEditor.remove(JsonField.KEY_USER_EMAIL);
51       mEditor.remove(JsonField.KEY_USER_PASSWORD);
52       mEditor.remove(JsonField.KEY_USER_MOBILENO);
53       mEditor.remove(IS_LOGIN);
54       mEditor.commit();
55     }
56   
57     public String getUserID()
58     {
59       return mSharedPreferences.getString(JsonField.KEY_USER_ID,"");
60     }
61   }
